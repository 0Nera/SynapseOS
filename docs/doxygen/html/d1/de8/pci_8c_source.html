35 uint32_t pci_read(pci_dev_t dev, uint32_t field) {
36  // Only most significant 6 bits of the field
37  dev.field_num = (field & 0xFC) >> 2;
38  dev.enable = 1;
39  outl(PCI_CONFIG_ADDRESS, dev.bits);
40 
41  // What size is this field supposed to be ?
42  uint32_t size = pci_size_map[field];
43  if(size == 1) {
44  // Get the first byte only, since it's in little endian, it's actually the 3rd byte
45  uint8_t t = inb(PCI_CONFIG_DATA + (field & 3));
46  return t;
47  }
48  else if(size == 2) {
49  uint16_t t = ins(PCI_CONFIG_DATA + (field & 2));
50  return t;
51  }
52  else if(size == 4){
53  // Read entire 4 bytes
54  uint32_t t = inl(PCI_CONFIG_DATA);
55  return t;
56  }
57  return 0xffff;
58 }

208 pci_dev_t pci_scan_function(uint16_t vendor_id, uint16_t device_id, uint32_t bus, uint32_t device, uint32_t function, int32_t device_type) {
209  pci_dev_t dev = {0};
210 
211  dev.bus_num = bus;
212  dev.device_num = device;
213  dev.function_num = function;
214  // If it's a PCI Bridge device, get the bus it's connected to and keep searching
215  if(get_device_type(dev) == PCI_TYPE_BRIDGE) {
216  pci_scan_bus(vendor_id, device_id, get_secondary_bus(dev), device_type);
217  }
218  // If type matches, we've found the device, just return it
219  if(device_type == -1 || device_type == get_device_type(dev)) {
220  uint32_t devid = pci_read(dev, PCI_DEVICE_ID);
221  uint32_t vendid = pci_read(dev, PCI_VENDOR_ID);
222  if(devid == device_id && vendor_id == vendid)
223  return dev;
224  }
225  return dev_zero;
226 }

236  pci_dev_t dev = {0};
237  dev.bus_num = bus;
238  dev.device_num = device;
239 
240  if(pci_read(dev,PCI_VENDOR_ID) == PCI_NONE)
241  return dev_zero;
242 
243  pci_dev_t t = pci_scan_function(vendor_id, device_id, bus, device, 0, device_type);
244  if(t.bits)
245  return t;
246 
247  if(pci_reach_end(dev))
248  return dev_zero;
249 
250  for(int32_t function = 1; function < FUNCTION_PER_DEVICE; function++) {
251  if(pci_read(dev,PCI_VENDOR_ID) != PCI_NONE) {
252  t = pci_scan_function(vendor_id, device_id, bus, device, function, device_type);
253  if(t.bits)
254  return t;
255  }
256  }
257  return dev_zero;
