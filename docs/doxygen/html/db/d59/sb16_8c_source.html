76void sb16_program_dma16(char channel, int address, short length) {
77 length -= 1;
78 outb(0xD4, 4+(channel%4)); // Channel
79 outb(0xD8, 1); // Flip-flop
80 outb(0xD6, 0b10110000); // 16-bit sound, no FIFO
81 char page = (address&0xFF0000)>>16; // 0x[AA]BBCC
82 char lop = address&0x0000FF; // 0xAABB[CC]
83 char hip = (address&0x00FF00)>>8; // 0xAA[BB]CC
84 outb(0x8B, page);
85 outb(0xC4, lop);
86 outb(0xC4, hip);
87 char lol = (length&0x00FF);
88 char hil = (length&0xFF00)>>8;
89 outb(0xC6, lol);
90 outb(0xC6, hil);
91 outb(0xD4, channel);
92}

125void sb16_play_audio(char *data, short sampling_rate, char channels, char eightbit, char sign, int length) {
126 // 1. Reset DSP
127 sb16_dsp_reset();
128 int loaded = 0;
129 int address = (void*)data;
130
131 // 3. Set master volume
132 sb16_set_master_volume(0xC, 0xC);
133 // 4. Turn speaker on
134 sb16_turn_speaker_on();
135 // 6, 7, 8, 9, 10
136 sb16_program(sampling_rate, channels, eightbit, sign, LOAD_LENGTH);
137
138 // Load sound data to memory
139 while(loaded<length-1) {
140 memcpy(driver_memory, data, LOAD_LENGTH);
141 sb16_program_dma16(channels, driver_memory, LOAD_LENGTH);
142 loaded+=LOAD_LENGTH;
143 data+=LOAD_LENGTH;
144 }
145}
