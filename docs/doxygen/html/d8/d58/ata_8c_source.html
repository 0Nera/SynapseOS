33 static uint8_t ide_read_register(uint8_t channel, uint8_t reg) {
34  uint8_t ret;
35 
36  // write value ata-control to tell irq is ready
37  if (reg > 0x07 && reg < 0x0C)
38  ide_write_register(channel, ATA_REG_CONTROL, 0x80 | g_ide_channels[channel].no_intr);
39 
40  // read register from base channel port
41  if (reg < 0x08)
42  ret = inb(g_ide_channels[channel].base + reg - 0x00);
43  else if (reg < 0x0C)
44  ret = inb(g_ide_channels[channel].base + reg - 0x06);
45  else if (reg < 0x0E)
46  ret = inb(g_ide_channels[channel].control + reg - 0x0A);
47  else if (reg < 0x16)
48  ret = inb(g_ide_channels[channel].bm_ide + reg - 0x0E);
49 
50  // write value to tell reading is done
51  if (reg > 0x07 && reg < 0x0C)
52  ide_write_register(channel, ATA_REG_CONTROL, g_ide_channels[channel].no_intr);
53 
54  return ret;
55 }

58 static void ide_write_register(uint8_t channel, uint8_t reg, uint8_t data) {
59  // write value ata-control to tell irq is ready
60  if (reg > 0x07 && reg < 0x0C)
61  ide_write_register(channel, ATA_REG_CONTROL, 0x80 | g_ide_channels[channel].no_intr);
62 
63  // write data to register ports

65  if(is_com_port(g_ide_channels[channel].base + reg - 0x00)){ return; }
66  outb(g_ide_channels[channel].base + reg - 0x00, data);
67  } else if (reg < 0x0C) {
68  if(is_com_port(g_ide_channels[channel].base + reg - 0x06)){ return; }
69  outb(g_ide_channels[channel].base + reg - 0x06, data);
70  } else if (reg < 0x0E) {
71  if(is_com_port(g_ide_channels[channel].control + reg - 0x0A)){ return; }
72  outb(g_ide_channels[channel].control + reg - 0x0A, data);
73  } else if (reg < 0x16) {
74  if(is_com_port(g_ide_channels[channel].bm_ide + reg - 0x0E)){ return; }
75  outb(g_ide_channels[channel].bm_ide + reg - 0x0E, data);
76  }
77  // write value to tell reading is done
78  if (reg > 0x07 && reg < 0x0C)
79  ide_write_register(channel, ATA_REG_CONTROL, g_ide_channels[channel].no_intr);

133 // wait until drive is ready, keep polling ide device until it is not busy status
134 uint8_t ide_polling(uint8_t channel, uint8_t advanced_check) {
135  // (I) Delay 400 nanosecond for BSY to be set:
136  for (int32_t i = 0; i < 4; i++)
137  // Reading the Alternate Status port wastes 100ns; loop four times.
138  ide_read_register(channel, ATA_REG_ALTSTATUS);
139 
140  // (II) Wait for BSY to be cleared:
141  while (ide_read_register(channel, ATA_REG_STATUS) & ATA_SR_BSY)
142  ; // Wait for BSY to be zero.
143 
144  if (advanced_check) {
145  // Read Status Register
146  uint8_t state = ide_read_register(channel, ATA_REG_STATUS);
147 
148  // (III) Check For Errors:
149  if (state & ATA_SR_ERR)
150  return 2; // Error.
151 
152  // (IV) Check If Device fault:
153  if (state & ATA_SR_DF)
154  return 1; // Device Fault.
155 
156  // (V) Check DRQ:
157  // BSY = 0; DF = 0; ERR = 0 so we should check for DRQ now.
158  if ((state & ATA_SR_DRQ) == 0)
159  return 3; // DRQ should be set
160  }
161 
162  return 0; // No Error.
163 }

229 void ide_init(uint32_t prim_channel_base_addr, uint32_t prim_channel_control_base_addr,
230  uint32_t sec_channel_base_addr, uint32_t sec_channel_control_addr,
231  uint32_t bus_master_addr) {
232  int32_t i, j, k, count = 0;
233  unsigned char ide_buf[2048] = {0};
234 
235  // 1- Detect I/O Ports which interface IDE Controller:
236  // (checking the addr is removed for simplicity, just assigning all ports)
237  g_ide_channels[ATA_PRIMARY].base = prim_channel_base_addr;
238  g_ide_channels[ATA_PRIMARY].control = prim_channel_control_base_addr;
239  g_ide_channels[ATA_SECONDARY].base = sec_channel_base_addr;
240  g_ide_channels[ATA_SECONDARY].control = sec_channel_control_addr;
241  g_ide_channels[ATA_PRIMARY].bm_ide = bus_master_addr;
242  g_ide_channels[ATA_SECONDARY].bm_ide = bus_master_addr;
243 
244  // 2- Disable IRQs:
245  ide_write_register(ATA_PRIMARY, ATA_REG_CONTROL, 2);
246  ide_write_register(ATA_SECONDARY, ATA_REG_CONTROL, 2);
247 
248  // 3- Detect ATA-ATAPI Devices:
249  for (i = 0; i < 2; i++) {
250  for (j = 0; j < 2; j++) {
251  uint8_t err = 0, type = IDE_ATA, status;
252  g_ide_devices[count].reserved = 0; // Assuming that no drive here.
253 
254  // (I) Select Drive:
255  ide_write_register(i, ATA_REG_HDDEVSEL, 0xA0 | (j << 4)); // Select Drive.
256  //sleep(1); // Wait 1ms for drive select to work.
257 
258  // (II) Send ATA Identify Command:
259  ide_write_register(i, ATA_REG_COMMAND, ATA_CMD_IDENTIFY);
260  //sleep(1); // This function should be implemented in your OS. which waits for 1 ms.
261  // it is based on System Timer Device Driver.
262 
263  // (III) Polling:
264  if (ide_read_register(i, ATA_REG_STATUS) == 0) continue; // If Status = 0, No Device.
265 
266  while (1) {
267  status = ide_read_register(i, ATA_REG_STATUS);
268  if ((status & ATA_SR_ERR)) {
269  err = 1;
270  break;
271  } // If Err, Device is not ATA.
272  if (!(status & ATA_SR_BSY) && (status & ATA_SR_DRQ)) break; // Everything is right.
273  }
274 
275  // (IV) Probe for ATAPI Devices:
276 
277  if (err != 0) {
278  unsigned char cl = ide_read_register(i, ATA_REG_LBA1);
279  unsigned char ch = ide_read_register(i, ATA_REG_LBA2);
280 
281  if (cl == 0x14 && ch == 0xEB)
282  type = IDE_ATAPI;
283  else if (cl == 0x69 && ch == 0x96)
284  type = IDE_ATAPI;
285  else
286  continue; // Unknown Type (may not be a device).
287 
288  ide_write_register(i, ATA_REG_COMMAND, ATA_CMD_IDENTIFY_PACKET);
289  //sleep(1);
290  }
291 
292  // (V) Read Identification Space of the Device:
293  ide_read_buffer(i, ATA_REG_DATA, (uint32_t *)ide_buf, 128);
294 
295  // (VI) Read Device Parameters:
296  g_ide_devices[count].reserved = 1;
297  g_ide_devices[count].type = type;
298  g_ide_devices[count].channel = i;
299  g_ide_devices[count].drive = j;
300  g_ide_devices[count].signature = *((unsigned short *)(ide_buf + ATA_IDENT_DEVICETYPE));
301  g_ide_devices[count].features = *((unsigned short *)(ide_buf + ATA_IDENT_CAPABILITIES));
302  g_ide_devices[count].command_sets = *((uint32_t *)(ide_buf + ATA_IDENT_COMMANDSETS));
303 
304  // (VII) Get Size:
305  if (g_ide_devices[count].command_sets & (1 << 26))
306  // Device uses 48-Bit Addressing:
307  g_ide_devices[count].size = *((uint32_t *)(ide_buf + ATA_IDENT_MAX_LBA_EXT));
308  else
309  // Device uses CHS or 28-bit Addressing:
310  g_ide_devices[count].size = *((uint32_t *)(ide_buf + ATA_IDENT_MAX_LBA));
311 
312  // (VIII) String indicates model of device (like Western Digital HDD and SONY DVD-RW...):
313  for (k = 0; k < 40; k += 2) {
314  g_ide_devices[count].model[k] = ide_buf[ATA_IDENT_MODEL + k + 1];
315  g_ide_devices[count].model[k + 1] = ide_buf[ATA_IDENT_MODEL + k];
316  }
317  g_ide_devices[count].model[40] = '\0'; // Terminate String.
318  // remove trailing spaces in model string
319  for(k = 39; k >= 0; k--) {
320  char ch = g_ide_devices[count].model[k];
321  if(ch == ' ')
322  g_ide_devices[count].model[k] = '\0';
323  else
324  break;
325  }
326 
327  count++;
328  }
329  }
330 
331  // 4- Print32_t Summary:
332  for (i = 0; i < 5; i++)
333  if (g_ide_devices[i].reserved == 1) {
334  tty_printf("%d:-\n", i);
335  tty_printf(" model: %s\n", g_ide_devices[i].model);
336  tty_printf(" type: %s\n", (const char *[]){"ATA", "ATAPI"}[g_ide_devices[i].type]);
337  tty_printf(" drive: %u, channel: %u\n", g_ide_devices[i].drive, g_ide_devices[i].channel);
338  tty_printf(" base: 0x%x, control: 0x%x\n", g_ide_channels[i].base, g_ide_channels[i].control);
339  tty_printf(" size: %u sectors, %u bytes\n", g_ide_devices[i].size, g_ide_devices[i].size * ATA_SECTOR_SIZE);
340  tty_printf(" signature: 0x%x, features: %d\n", g_ide_devices[i].signature, g_ide_devices[i].features);
341  }
342 }

344 uint8_t ide_ata_access(uint8_t direction, uint8_t drive, uint32_t lba, uint8_t num_sectors, uint32_t buffer) {
345  uint8_t lba_mode /* 0: CHS, 1:LBA28, 2: LBA48 */, dma /* 0: No DMA, 1: DMA */, cmd;
346  uint8_t lba_io[6];
347  uint32_t channel = g_ide_devices[drive].channel; // Read the Channel.
348  uint32_t slavebit = g_ide_devices[drive].drive; // Read the Drive [Master/Slave]
349  uint32_t bus = g_ide_channels[channel].base; // Bus Base, like 0x1F0 which is also data port.
350  uint32_t words = 256; // Almost every ATA drive has a sector-size of 512-byte.
351  uint16_t cyl, i;
352  uint8_t head, sect, err;
353 
354  ide_write_register(channel, ATA_REG_CONTROL, g_ide_channels[channel].no_intr = (g_ide_irq_invoked = 0x0) + 0x02);
355 
356  // (I) Select one from LBA28, LBA48 or CHS;
357  if (lba >= 0x10000000) { // Sure Drive should support LBA in this case, or you are
358  // giving a wrong LBA.
359  // LBA48:
360  lba_mode = LBA_MODE_48;
361  lba_io[0] = (lba & 0x000000FF) >> 0;
362  lba_io[1] = (lba & 0x0000FF00) >> 8;
363  lba_io[2] = (lba & 0x00FF0000) >> 16;
364  lba_io[3] = (lba & 0xFF000000) >> 24;
365  lba_io[4] = 0; // LBA28 is integer, so 32-bits are enough to access 2TB.
366  lba_io[5] = 0; // LBA28 is integer, so 32-bits are enough to access 2TB.
367  head = 0; // Lower 4-bits of HDDEVSEL are not used here.
368  } else if (g_ide_devices[drive].features & 0x200) { // Drive supports LBA?
369  // LBA28:
370  lba_mode = LBA_MODE_28;
371  lba_io[0] = (lba & 0x00000FF) >> 0;
372  lba_io[1] = (lba & 0x000FF00) >> 8;
373  lba_io[2] = (lba & 0x0FF0000) >> 16;
374  lba_io[3] = 0; // These Registers are not used here.
375  lba_io[4] = 0; // These Registers are not used here.
376  lba_io[5] = 0; // These Registers are not used here.
377  head = (lba & 0xF000000) >> 24;
378  } else {
379  // CHS:
380  lba_mode = LBA_MODE_CHS;
381  sect = (lba % 63) + 1;
382  cyl = (lba + 1 - sect) / (16 * 63);
383  lba_io[0] = sect;
384  lba_io[1] = (cyl >> 0) & 0xFF;
385  lba_io[2] = (cyl >> 8) & 0xFF;
386  lba_io[3] = 0;
387  lba_io[4] = 0;
388  lba_io[5] = 0;
389  head = (lba + 1 - sect) % (16 * 63) / (63); // Head number is written to HDDEVSEL lower 4-bits.
390  }
391 
392  // (II) See if drive supports DMA or not;
393  dma = 0; // We don't support DMA
394 
395  // (III) Wait if the drive is busy;
396  while (ide_read_register(channel, ATA_REG_STATUS) & ATA_SR_BSY) {
397  } // Wait if busy.
398 
399  // (IV) Select Drive from the controller;
400  if (lba_mode == LBA_MODE_CHS)
401  ide_write_register(channel, ATA_REG_HDDEVSEL, 0xA0 | (slavebit << 4) | head); // Drive & CHS.
402  else
403  ide_write_register(channel, ATA_REG_HDDEVSEL, 0xE0 | (slavebit << 4) | head); // Drive & LBA
404 
405  // (V) Write Parameters;
406  if (lba_mode == LBA_MODE_48) {
407  ide_write_register(channel, ATA_REG_SECCOUNT1, 0);
408  ide_write_register(channel, ATA_REG_LBA3, lba_io[3]);
409  ide_write_register(channel, ATA_REG_LBA4, lba_io[4]);
410  ide_write_register(channel, ATA_REG_LBA5, lba_io[5]);
411  }
412  ide_write_register(channel, ATA_REG_SECCOUNT0, num_sectors);
413  ide_write_register(channel, ATA_REG_LBA0, lba_io[0]);
414  ide_write_register(channel, ATA_REG_LBA1, lba_io[1]);
415  ide_write_register(channel, ATA_REG_LBA2, lba_io[2]);
416 
417  // (VI) Select the command and send it;
418  if (lba_mode == LBA_MODE_CHS && dma == 0 && direction == ATA_READ) cmd = ATA_CMD_READ_PIO;
419  if (lba_mode == LBA_MODE_28 && dma == 0 && direction == ATA_READ) cmd = ATA_CMD_READ_PIO;
420  if (lba_mode == LBA_MODE_48 && dma == 0 && direction == ATA_READ) cmd = ATA_CMD_READ_PIO_EXT;
421  if (lba_mode == LBA_MODE_CHS && dma == 1 && direction == ATA_READ) cmd = ATA_CMD_READ_DMA;
422  if (lba_mode == LBA_MODE_28 && dma == 1 && direction == ATA_READ) cmd = ATA_CMD_READ_DMA;
423  if (lba_mode == LBA_MODE_48 && dma == 1 && direction == ATA_READ) cmd = ATA_CMD_READ_DMA_EXT;
424  if (lba_mode == LBA_MODE_CHS && dma == 0 && direction == ATA_WRITE) cmd = ATA_CMD_WRITE_PIO;
425  if (lba_mode == LBA_MODE_28 && dma == 0 && direction == ATA_WRITE) cmd = ATA_CMD_WRITE_PIO;
426  if (lba_mode == LBA_MODE_48 && dma == 0 && direction == ATA_WRITE) cmd = ATA_CMD_WRITE_PIO_EXT;
427  if (lba_mode == LBA_MODE_CHS && dma == 1 && direction == ATA_WRITE) cmd = ATA_CMD_WRITE_DMA;
428  if (lba_mode == LBA_MODE_28 && dma == 1 && direction == ATA_WRITE) cmd = ATA_CMD_WRITE_DMA;
429  if (lba_mode == LBA_MODE_48 && dma == 1 && direction == ATA_WRITE) cmd = ATA_CMD_WRITE_DMA_EXT;
430  ide_write_register(channel, ATA_REG_COMMAND, cmd); // Send the Command.
431 
432  if (dma) {
433  tty_printf("DMA!!!");
434  if (direction == ATA_READ) {
435  // DMA Read
436 
437  } else {
438  // DMA write
439  }
440  } else if (direction == ATA_READ) {
441  // PIO Read.
442  for (i = 0; i < num_sectors; i++) {
443  if ((err = ide_polling(channel, 1)))
444  return err; // Polling, set error and exit if there is.
445 
446  // save es segment and repeat insw(read stream of shorts) instruction util no of sectors are read into buffer
447  asm volatile("pushw %es");
448  asm volatile("rep insw"
449  :
450  : "c"(words), "d"(bus), "D"(buffer)); // Receive Data.
451  asm volatile("popw %es");
452  buffer += (words * 2);
453  }
454  } else {
455  // PIO Write.
456  for (i = 0; i < num_sectors; i++) {
457  ide_polling(channel, 0); // Polling.
458  // save es segment and repeat outsw(write stream of shorts) instruction util no of sectors are written to ide device
459  asm volatile("pushw %ds");
460  asm volatile("rep outsw" ::"c"(words), "d"(bus), "S"(buffer)); // Send Data
461  asm volatile("popw %ds");
462  buffer += (words * 2);
463  }
464  // send the flush commands
465  ide_write_register(channel, ATA_REG_COMMAND, (char[]){ATA_CMD_CACHE_FLUSH, ATA_CMD_CACHE_FLUSH, ATA_CMD_CACHE_FLUSH_EXT}[lba_mode]);
466  ide_polling(channel, 0); // Polling.
467  }
468 
469  return 0; // Easy, isn't it?
470 }

482 int32_t ide_read_sectors(uint8_t drive, uint8_t num_sectors, uint32_t lba, uint32_t buffer) {
483  // 1: Check if the drive presents:
484  if (drive > MAXIMUM_IDE_DEVICES || g_ide_devices[drive].reserved == 0) {
485  tty_printf("IDE ERROR: Drive not found\n");
486  return -1;
487  }

579 int ata_read_one_sector_dma(struct ata_dma_priv *adp, char *buf, size_t lba) {
580  uint16_t io = ATA_PRIMARY_IO;
581  uint8_t dr = ATA_MASTER;
582  /* XXX: io & dr need to be dynamic once multiple ATA devices
583  * are implemented
584  */
585 
586  uint8_t cmd = 0xE0;
587  int errors = 0;
588  uint8_t slavebit = 0x00;
589 
590  ata_wait(io, 0);
591 
592  /* set up DMA transfer by sending STOP */
593  outb(adp->adp_busmaster, 0x00);
594 
595  /* send the PRDT */
596  outl(adp->adp_busmaster + 0x04, kv2p(adp->adp_dma_prdt));
597 
598  /* enable ERR & IRQ status */
599  outb(adp->adp_busmaster + 0x02,
600  inb(adp->adp_busmaster + 0x02) | 0x04 | 0x02);
601 
602  /* set direction */
603  outb(adp->adp_busmaster, 0x08);
604 
605  /* wait till the device is ready */
606  while (1) {
607  uint8_t status = inb(io + ATA_REG_STATUS);
608  if (!(status & ATA_SR_BSY)) break;
609  }
610 
611  //printk("ata: lba: %d\n", lba);
612  outb(io + ATA_REG_CONTROL, 0x00);
613 
614  outb(io + ATA_REG_HDDEVSEL, (cmd | (uint8_t)((lba >> 24 & 0x0F))));
615  ata_io_wait(io);
616  outb(io + ATA_REG_FEATURES, 0x00);
617  outb(io + ATA_REG_SECCOUNT0, 1);
618  outb(io + ATA_REG_LBA0, (uint8_t)(lba));
619  outb(io + ATA_REG_LBA1, (uint8_t)(lba >> 8));
620  outb(io + ATA_REG_LBA2, (uint8_t)(lba >> 16));
621 
622  /* wait again */
623  //printk("about to wait 1\n");
624  while (1) {
625  uint8_t status = inb(io + ATA_REG_STATUS);
626  if (!(status & ATA_SR_BSY) && (status & ATA_SR_DRDY)) break;
627  }
628  //printk("finished wait 1\n");
629 
630  outb(io + ATA_REG_COMMAND, ATA_CMD_READ_DMA);
631 
632  //printk("about to wait 2\n");
633  ata_io_wait(io);
634  //printk("finished wait 2\n");
635 
636  outb(adp->adp_busmaster, 0x08 | 0x01);
637 
638  //printk("about to wait 3\n");
639  while (1) {
640  int status = inb(adp->adp_busmaster + 0x02);
641  int dstatus = inb(io + ATA_REG_STATUS);
642  if (!(status & 0x04)) {
643  continue;
644  }
645  if (!(dstatus & ATA_SR_BSY)) {
646  break;
647  }
648  }
649  //printk("finished wait 3\n");
650 
651  memcpy(buf, adp->adp_dma_area, 512);
652 
653  outb(adp->adp_busmaster+ 0x2, inb(adp->adp_busmaster + 0x02) | 0x04 | 0x02);
654 
655  return 0;
656 }
