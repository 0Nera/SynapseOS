165void qemu_putuint(uint32_t i) {
166 uint32_t n, d = 1000000000;
167 char str[255];
168 uint32_t dec_index = 0;
169
170 while ((i / d == 0) && (d >= 10)) {
171 d /= 10;
172 }
173 n = i;
174
175 while (d >= 10) {
176 str[dec_index++] = ((char) ((int) '0' + n / d));
177 n = n % d;
178 d /= 10;
179 }
180
181 str[dec_index++] = ((char) ((int) '0' + n));
182 str[dec_index] = 0;
183 qemu_putstring(str);
184}

257void qemu_print(char *format, va_list args) {
258 int32_t i = 0;
259 char *string;
260
261 while (format[i]) {
262 if (format[i] == '%') {
263 i++;
264 switch (format[i]) {
265 case 's':
266 string = va_arg(args, char*);
267 qemu_putstring(string);
268 break;
269 case 'c':
270 // To-Do: fix this! "warning: cast to pointer from integer of different size"
271 com1_write_char((char)va_arg(args, int));
272 break;
273 case 'd':
274 qemu_putint(va_arg(args, int));
275 break;
276 case 'i':
277 qemu_putint(va_arg(args, int));
278 break;
279 case 'u':
280 qemu_putuint(va_arg(args, unsigned int));
281 break;
282 case 'x':
283 qemu_puthex(va_arg(args, uint32_t));
284 break;
285 case 'v':
286 qemu_puthex_v(va_arg(args, uint32_t));
287 break;
288 default:
289 com1_write_char(format[i]);
290 }
291 } else {
292 com1_write_char(format[i]);
293 }
294 i++;
295 }
296}
