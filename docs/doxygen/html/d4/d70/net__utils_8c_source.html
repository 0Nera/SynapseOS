9uint32_t flip_long(uint32_t long_int) {
10 uint32_t first_byte = *((uint8_t*)(&long_int));
11 uint32_t second_byte = *((uint8_t*)(&long_int) + 1);
12 uint32_t third_byte = *((uint8_t*)(&long_int) + 2);
13 uint32_t fourth_byte = *((uint8_t*)(&long_int) + 3);
14 return (first_byte << 24) | (second_byte << 16) | (third_byte << 8) | (fourth_byte);
15}

17/*
18 * Flip two parts within a byte
19 * For example, 0b11110000 will be 0b00001111 instead
20 * This is necessary because endiness is also relevant to byte, where there are two fields in one byte.
21 * number_bits: number of bits of the less significant field
22 * */
23uint8_t flip_byte(uint8_t byte, int num_bits) {
24 uint8_t t = byte << (8 - num_bits);
25 return t | (byte >> num_bits);
26}
