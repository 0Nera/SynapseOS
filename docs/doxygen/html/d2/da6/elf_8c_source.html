10// Returns 0 if header is valid, 1 if magic number invalid, 2 and more if file isn't compatible
11uint8_t elf_check_header(struct elf_hdr *hdr) {
12 if (hdr->mag_num[0] != 0x7f ||
13 hdr->mag_num[1] != 'E' ||
14 hdr->mag_num[2] != 'L' ||
15 hdr->mag_num[3] != 'F') {
16 return 1; // Our magic number isn't valid
17 }
18 if (hdr->arch != ELF_ARCH_32BIT) {
19 return 2; // This ELF file can't be loaded because it's not 32-bit
20 }
21 if (hdr->byte_order != ELF_BYTEORDER_LENDIAN) {
22 return 3; // This ELF file can't be loaded because it's not little-endian
23 }
24 if (hdr->elf_ver != 1) {
25 return 4; // This ELF file can't be loaded because its version isn't 1
26 }
27 if (hdr->file_type != ELF_REL && hdr->file_type != ELF_EXEC) {
28 return 5; // This ELF file can't be loaded because it's neither executable nor relocatable file
29 }
30 if (hdr->machine!=ELF_386_MACHINE) {
31 return 6; // This ELF file can't be loaded because it's not for i386 platform
32 }
33 return 0;
34}

38 if (!vfs_exists(fname)) {

44 uint32_t fsize = vfs_get_size(fname);
45 void *addr = kheap_malloc(fsize);
46 int32_t res = vfs_read(fname, 0, fsize, addr);

48 qemu_log("elf_open res = %d", res);
49
50 struct elf_hdr *hdr = addr;

174int32_t run_elf_file(const char *name/*, char **argv, char **env __attribute__((unused)), int32_t argc*/) {
175 if (!vfs_exists(name)) {
176 tty_printf("\nrun_elf_file: elf [%s] does not exist\n", name);
177 qemu_log("run_elf_file: elf [%s] does not exist", name);
178 return 0;
179 }
180 void *elf_file = elf_open(name);
181 if (elf_file == NULL) {
182 return -1;
183 }
184
185 struct elf_hdr *hdr = (struct elf_hdr*) elf_file;
186
187 qemu_log("loading segments:");
188 uint32_t vmm_alloced[4096] = {0};
189 int32_t ptr_vmm_alloced = 0;
190
191 for (int32_t i = 0; i < hdr->ph_ent_cnt; i++) {
192 //printf("Segment [%i/%i]: ", i, hdr->ph_ent_cnt);
193 elf_program_header_t *phdr = elf_get_program_header(elf_file, i);
194 if (phdr->type != SEGTYPE_LOAD) {
195 continue; // We only can load segments to the memory, so just skip it.
196 }
197
198 qemu_log("Loading %x bytes to %x", phdr->size_in_mem, phdr->load_to);
199
200 // Alloc needed amount of pages
201 uint32_t alloc_addr;
202 for (alloc_addr = phdr->load_to;
203 alloc_addr < phdr->load_to + phdr->size_in_mem;
204 alloc_addr += PAGE_SIZE) {
205 vmm_alloced[ptr_vmm_alloced] = alloc_addr;
206 ptr_vmm_alloced++;
207 qemu_log("Alloc %d: %x", ptr_vmm_alloced, alloc_addr);
208 vmm_alloc_page(alloc_addr);
209 }
210
211 memset((void*) phdr->load_to, 0, phdr->size_in_mem); // Null segment memory
212 memcpy((void*) phdr->load_to, elf_file + phdr->data_offset, phdr->size_in_file);
213 qemu_log("Loaded");
214 }
215
216 int(*entry_point)() = (void*) (hdr->entry);
217 qemu_log("ELF entry point: %x", hdr->entry);
218
219 qemu_log("Executing");
220 int _result = entry_point();
221
222 tty_printf("\n[PROGRAMM FINISHED WITH CODE <%d>]", _result);
223 qemu_log("[PROGRAMM FINISHED WITH CODE <%d>]", _result);
224 qemu_log("Cleaning VMM:");
225
226 for (int32_t i = 0; i != ptr_vmm_alloced; i++){
227 qemu_log("\tCleaning %d: %x", i, vmm_alloced[i]);
228 vmm_free_page(vmm_alloced[i]);
229 }
230 qemu_log("[CLEANED <%d> PAGES]", ptr_vmm_alloced);
231
232 return 0;
233}
