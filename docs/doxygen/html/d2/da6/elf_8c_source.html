10 // Returns 0 if header is valid, 1 if magic number invalid, 2 and more if file isn't compatible
11 uint8_t elf_check_header(struct elf_hdr *hdr) {
12  if (hdr->mag_num[0] != 0x7f ||
13  hdr->mag_num[1] != 'E' ||
14  hdr->mag_num[2] != 'L' ||
15  hdr->mag_num[3] != 'F') {
16  return 1; // Our magic number isn't valid
17  }
18  if (hdr->arch != ELF_ARCH_32BIT) {
19  return 2; // This ELF file can't be loaded because it's not 32-bit
20  }
21  if (hdr->byte_order != ELF_BYTEORDER_LENDIAN) {
22  return 3; // This ELF file can't be loaded because it's not little-endian
23  }
24  if (hdr->elf_ver != 1) {
25  return 4; // This ELF file can't be loaded because its version isn't 1
26  }
27  if (hdr->file_type != ELF_REL && hdr->file_type != ELF_EXEC) {
28  return 5; // This ELF file can't be loaded because it's neither executable nor relocatable file
29  }
30  if (hdr->machine!=ELF_386_MACHINE) {
31  return 6; // This ELF file can't be loaded because it's not for i386 platform
32  }
33  return 0;
34 }

181 int32_t run_elf_file(const char *name, int32_t argc, char **argv) {
182  if (!vfs_exists(name)) {
183  tty_printf("\nrun_elf_file: elf [%s] does not exist\n", name);
184  qemu_log("run_elf_file: elf [%s] does not exist", name);
185  return 0;
186  }
187  void *elf_file = elf_open(name);
188  if (elf_file == NULL) {
189  return -1;
190  }
191 
192  struct elf_hdr *hdr = (struct elf_hdr*) elf_file;
193 
194  qemu_log("loading segments:");
195  uint32_t vmm_alloced[4096] = {0};
196  int32_t ptr_vmm_alloced = 0;
197 
198  for (int32_t i = 0; i < hdr->ph_ent_cnt; i++) {
199  //printf("Segment [%i/%i]: ", i, hdr->ph_ent_cnt);
200  elf_program_header_t *phdr = elf_get_program_header(elf_file, i);
201  if (phdr->type != SEGTYPE_LOAD) {
202  continue; // We only can load segments to the memory, so just skip it.
203  }
204 
205  qemu_log("Loading %x bytes to %x", phdr->size_in_mem, phdr->load_to);
206 
207  // Alloc needed amount of pages
208  uint32_t alloc_addr;
209  for (alloc_addr = phdr->load_to;
210  alloc_addr < phdr->load_to + phdr->size_in_mem;
211  alloc_addr += PAGE_SIZE) {
212  vmm_alloced[ptr_vmm_alloced] = alloc_addr;
213  ptr_vmm_alloced++;
214  qemu_log("Alloc %d: %x", ptr_vmm_alloced, alloc_addr);
215  vmm_alloc_page(alloc_addr);
216  }
217 
218  memset((void*) phdr->load_to, 0, phdr->size_in_mem); // Null segment memory
219  memcpy((void*) phdr->load_to, elf_file + phdr->data_offset, phdr->size_in_file);
220  qemu_log("Loaded");
221  }
222 
223  int(*entry_point)(int argc, char** argv) = (void*) (hdr->entry);
224  qemu_log("ELF entry point: %x", hdr->entry);
225 
226  qemu_log("Executing");
227  int _result = entry_point(argc, argv);
228 
229  tty_printf("\n[PROGRAMM FINISHED WITH CODE <%d>]", _result);
230  qemu_log("[PROGRAMM FINISHED WITH CODE <%d>]", _result);
231  qemu_log("Cleaning VMM:");
232 
233  for (int32_t i = 0; i != ptr_vmm_alloced; i++){
234  qemu_log("\tCleaning %d: %x", i, vmm_alloced[i]);
235  vmm_free_page(vmm_alloced[i]);
236  }
237  qemu_log("[CLEANED <%d> PAGES]", ptr_vmm_alloced);
238 
239  return 0;
240 }
