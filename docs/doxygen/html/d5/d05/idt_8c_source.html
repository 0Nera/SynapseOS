121void pic_init(void) {
122 /* Send ICW1 to both PIC chips */
123 outb(PIC1_COMMAND, ICW1_INIT | ICW1_CALLADDR_4 | ICW1_ICW4_NEEDED);
124 outb(PIC2_COMMAND, ICW1_INIT | ICW1_CALLADDR_4 | ICW1_ICW4_NEEDED);
125
126 /* Determine the offset */
127 outb(PIC1_DATA, 32);
128 outb(PIC2_DATA, 40);
129
130 /* Configure active/passive device cascade */
131 outb(PIC1_DATA, 4);
132 outb(PIC2_DATA, 2);
133
134 /* Set operation mode - 8086 mode on*/
135 outb(PIC1_DATA, 1);
136 outb(PIC2_DATA, 1);
137
138 /* mask (disable) all PIC interrupts */
139 pic_disable_all_irq();
140
141 qemu_log("PIC installed");
142}

296void irq_handler(struct regs *r) {
297 // Blank function pointer
298 void (*handler)(struct regs *r);
299
300 // If there's a custom handler to handle the IRQ, handle it
301 handler = interrupt_handlers[r->idt_index];
302 if (handler) {
303 handler(r);
304 }
305
306 // If the IDT entry that was invoked was greater than 40, sends an EOI
307 // to the slave controller
308 if (r->idt_index >= 40) {
309 outb(0xA0, 0x20);
310 }
311
312 // Sends an EOI to the master interrupt controller
313 outb(0x20, 0x20);
314}

316void run_interrupt_handler(struct regs* r) {
317 size_t idt_index = r->idt_index;
318 if (idt_index < 32) {
319 fault_handler(r);
320 return;
321 }
322
323 if (idt_index >= 32 && idt_index <= 47) {
324 irq_handler(r);
325 return;
326 }
327
328 if (interrupt_handlers[r->idt_index] != NULL) {
329 interrupt_handlers[r->idt_index](r);
330 }
