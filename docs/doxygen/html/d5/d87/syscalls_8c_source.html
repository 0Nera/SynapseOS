34 void syscall_handler(struct regs *r) {
35  /*
36  Регистры сисфункций:
37  eax - номер сисфункции и результат
38  ebx - параметр-указатель 1
39  edx - параметр-указатель 2
40  ecx - параметр-указатель 3
41  esi - параметр-указатель 4
42  edi - параметр-указатель 5
43  */
44  uint32_t arg1 = r->ebx;
45  uint32_t arg2 = r->edx;
46  uint32_t arg3 = r->ecx;
47  uint32_t arg4 = r->esi;
48  uint32_t arg5 = r->edi;
49 
50  switch (r->eax) {
51  case SC_CODE_puts: // I/O
52  tty_printf("%s", (char *)(arg1));
53  r->eax = (uint32_t)1;
54  break;
55  case SC_CODE_TTY_DRAW_RECT:
56  drawRect((int)(arg1),(int)(arg2),(int)(arg3),(int)(arg4),(int)(arg5));
57  r->eax = (uint32_t)1;
58  break;
59  case SC_CODE_CHANGE_KEYBOARD:
60  changeStageKeyboard((int32_t)(arg1));
61  r->eax = (uint32_t)1;
62  break;
63  case SC_CODE_GET_KEY_LAST:
64  r->eax = (uint32_t)keyLastInset();
65  break;
66  case SC_CODE_TTY_POS_X:
67  r->eax = (uint32_t)getPosX();
68  break;
69  case SC_CODE_TTY_POS_Y:
70  r->eax = (uint32_t)getPosY();
71  break;
72  case SC_CODE_TTY_SCREEN_W:
73  r->eax = (uint32_t)getWidthScreen();
74  break;
75  case SC_CODE_TTY_SCREEN_H:
76  r->eax = (uint32_t)getHeightScreen();
77  break;
78  case SC_CODE_TTY_SET_COLOR:
79  tty_setcolor((int32_t)(arg1));
80  r->eax = (uint32_t)1;
81  break;
82  case SC_CODE_TTY_SET_POS_X:
83  setPosX((int32_t)(arg1));
84  r->eax = (uint32_t)1;
85  break;
86  case SC_CODE_TTY_SET_POS_Y:
87  setPosY((int32_t)(arg1));
88  r->eax = (uint32_t)1;
89  break;
90  case SC_CODE_TTY_PUT_COLOR:
91  tty_putchar_color((char)(arg1),(uint32_t)(arg2),(uint32_t)(arg3));
92  r->eax = (uint32_t)1;
93  break;
94  case SC_CODE_TTY_PUTS_COLOR:
95  tty_puts_color((char *)(arg1),(uint32_t)(arg2),(uint32_t)(arg3));
96  r->eax = (uint32_t)1;
97  break;
98  case SC_CODE_TTY_DRAW_VGA:
99  draw_vga_character((uint8_t)(arg1), (int32_t)(arg2), (int32_t)(arg3), (int32_t)(arg4), (int32_t)(arg5), true);
100  break;
101 
102  case SC_CODE_getscancode:
103  r->eax = (uint32_t)keyboard_getscancode();
104  break;
105  case SC_CODE_getchar:
106  r->eax = (uint32_t)keyboard_getchar();
107  break;
108  case SC_CODE_gets:
109  r->eax = (uint32_t)keyboard_gets();
110  break;
111  case SC_CODE_malloc: {
112  r->eax = (uint32_t)kheap_malloc((uint32_t)arg1);
113  //tty_printf("Now memory (%d) in: %x\n", arg1, r->eax);
114  break;
115  }
116  case SC_CODE_free: {
117  kheap_free((void*)arg1);
118  r->eax = (uint32_t)1;
119  break;
120  }
121  case SC_CODE_setdev: // Хранилище
122  r->eax = (uint32_t)-1;
123  break;
124  case SC_CODE_readfile:
125  r->eax = (uint32_t)vfs_read((char *)arg1, (int32_t)arg2, (int32_t)arg3, (void *)arg4);
126  //tty_printf("readfile(): got arguments: %s, %d, %d, %x\n", arg1, arg2, arg3, arg4);
127  break;
128  case SC_CODE_exists:
129  r->eax = (uint32_t)vfs_exists((char *)arg1);
130  break;
131  case SC_CODE_filesize:
132  r->eax = (uint32_t)vfs_get_size((char*)arg1);
133  break;
134  case SC_CODE_rand:
135  r->eax = (uint64_t)rand();
136  break;
137  case SC_CODE_srand:
138  srand((uint64_t)(arg1));
139  r->eax = (uint64_t)rand();
140  break;
141  case SC_CODE_putpixel: // Графика
142  set_pixel((int32_t)(arg1), (int32_t)(arg2), (uint32_t)(arg3));
143  r->eax = (uint32_t)1;
144  break;
145  case SC_CODE_drawline:
146  set_line((int32_t)(arg1), (int32_t)(arg2), (int32_t)(arg3), (int32_t)(arg4), (uint32_t)(arg5));
147  r->eax = (uint32_t)1;
148  break;
149  case SC_CODE_version: // Система
150  r->eax = (uint32_t)(VERSION_MAJOR * 100 + VERSION_MINOR);
151  break;
152  case SC_CODE_newtask:
153  r->eax = add_task(&r);
154  break;
155  case SC_CODE_ticks:
156  r->eax = timer_get_ticks();
157  break;
158  case SC_CODE_getfreq:
159  r->eax = timer_get_frequency();
160  break;
161  case SC_CODE_getttyinfo:
162  r->eax = get_tty_info();
163  break;
164  case SC_CODE_get_time:
165  r->eax = get_time_pointer();
166  break;
167  default:
168  qemu_log("Invalid syscall #%x", r->eax);
169  qemu_log("r->idt_index = %x eax = %x ebx = %x "
170  "ecx = %x edx = %x esp = %x ebp = %x eip = %x",
171  r->idt_index, r->eax, r->ebx,
172  r->ecx, r->edx, r->esp, r->ebp, r->eip);
173  tty_printf("Invalid syscall %x\n", r->eax);
174  tty_printf("Invalid syscall %d\n", r->eax);
175  tty_printf("r->idt_index = %x eax = %x ebx = %x "
176  "ecx = %x edx = %x esp = %x ebp = %x eip = %x",
177  r->idt_index, r->eax, r->ebx,
178  r->ecx, r->edx, r->esp, r->ebp, r->eip);
179  tty_printf("r->idt_index = %d eax = %d ebx = %d "
180  "ecx = %d edx = %d esp = %d ebp = %d eip = %d",
181  r->idt_index, r->eax, r->ebx,
182  r->ecx, r->edx, r->esp, r->ebp, r->eip);
183  }
184 }
