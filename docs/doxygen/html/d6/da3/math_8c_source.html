64 float powf( float x, float y )
65 {
66  static const double recip_ln2 = 0x1.71547652b82fep0;
67 
68  if( x == 1.0f || y == 1.0f)
69  return x;
70 
71  //Move the arguments to the integer registers for bitwise inspection
72  union{ float f; uint32_t u; } ux, uy;
73  ux.f = x;
74  uy.f = y;
75  uint32_t absux = ux.u & 0x7fffffff;
76  uint32_t absuy = uy.u & 0x7fffffff;
77 
78 
79  // Handle most edge cases
80  //If |x| or |y| is in { +-0, +-Inf, +-NaN }
81  if( (ux.u - 1U) >= 0x7f7fffff || (absuy - 1) >= 0x4affffff )
82  {
83  // any**0 = 1.0f for all values, including NaN
84  if( 0 == absuy )
85  return 1.0f;
86 
87  // handle NaNs
88  if( x != x || y != y )
89  return x + y;
90 
91  //figure out if y is an odd integer
92  //Find out if y is an integer or not without raising inexact
93  // Note -- independently tested over entire range. Fails for Inf/NaN. We don't care about that here.
94  uint32_t fractMask = 0x3fffffffU >> gMaskShift[ absuy >> 23 ]; //mask bits cover fractional part of value
95  uint32_t onesMask = 0x40000000U >> gMaskShift[ absuy >> 23 ]; // we get away with this because leading exponent bit is never set for |y| < 2.0
96  uint32_t fractionalBits = absuy & fractMask;
97  uint32_t onesBit = absuy & onesMask;
98 
99  if( 0 == absux )
100  {
101  //if y is an odd integer
102  if( 0 == fractionalBits && 0 != onesBit )
103  {
104  if( y < 0.0f )
105  return 1.0f / x;
106 
107  return x;
108  }
109 
110  // y is not an odd integer
111  if( 0.0f < y )
112  return 0.0f;
113 
114  return 1.0f / __builtin_fabsf(x); // return Inf and set div/0
115 
116  }
117 
118  // deal with infinite y
119  if( 0x7f800000 == absuy )
120  {
121  if( -1.0f == x )
122  return 1.0f;
123 
124  if( absux > 0x3f800000 ) // |x| > 1.0f
125  { // |x| > 1.0f
126  if( 0.0f < y )
127  return y;
128  else
129  return 0.0f;
130  }
131  else
132  { // |x| < 1.0f
133  if( 0.0f < y )
134  return 0.0f;
135  else
136  return __builtin_fabsf(y);
137  }
138  }
139 
140  // we can also deal with x == +inf at this point.
141  if( x == __builtin_inff() )
142  {
143  if( y < 0.0f )
144  return 0.0f;
145  else
146  return x;
147  }
148 
149  if( x > -__builtin_inff() )
150  {
151  if( fractionalBits )
152  goto nan_sqrt;
153 
154  goto ipowf;
155  }
156 
157  // At this point, we know that x is in { +-0, -Inf } and y is finite non-zero.
158  // Deal with y is odd integer cases
159  if( 0 == fractionalBits && 0 != onesBit ) // if( |y| >= 1.0f || |y| < 0x1.0p24f )
160  return 0.0f < y ? x : -0.0f;
161 
162  // x == -inf
163  return 0.0f < y ? -x : 0.0f;
164  }
165 
166  //special case for sqrts
167  if( 0x3f000000U == absuy )
168  goto nan_sqrt;
169 
170  // break |x| into exponent and fractional parts: |x| = 2**i * m 1.0 <= m < 2.0
171  int32_t i = ((absux >> 23) & 0xff) - 127;
172  union
173  {
174  uint32_t u;
175  float f;
176  }m = { (absux & 0x007fffffU) | 0x3f800000U };
177 
178  //normalize denormals
179  if( -127 == i )
180  { //denormal
181  m.f -= 1.0f; // exact
182  i = ((m.u >> 23) & 0xff) - (127+126);
183  m.u = (m.u & 0x807fffffU) | 0x3f800000U;
184  }
185 
186  //
187  // We further break down m as :
188  //
189  // m = (1+a/256.0)(1+r) a = high 8 explicit bits of mantissa(m), b = next 7 bits
190  // log2f(m) = log2(1+a/256.0) + log2(1+r)
191  //
192  // We use the high 7 bits of the mantissa to look up log2(1+a/256.0) in log2f_table above
193  // We calculate 1+r as:
194  //
195  // 1+r = m * (1 /(1+a/256.0))
196  //
197  // We can lookup (from the same table) the value of 1/(1+a/256.0) based on a too.
198 
199  double log2x = i;
200 
201  if( m.f != 1.0f )
202  {
203  int index = (m.u >> (23-7-4)) & 0x7f0; //top 7 bits of mantissa
204  const double *tablep = (void*) powf_log_table + index;
205  double r = (double) m.f;
206 
207  // reduce
208  r *= tablep[0]; // reduce r to 1-2**-7 < r < 1+2**-7
209  log2x += tablep[1]; // do this early to force -1.0 + 1.0 to cancel so that we don't end up with (1.0 + tiny) - 1.0 later on.
210  r -= 1.0; // -2**-7 < r < 1+2**-7
211 
212  // ln(1+r) = r - rr/2 + rrr/3 - rrrr/4 + rrrrr/5
213  // should provide log(1+r) to at least 35 bits of accuracy for the worst case
214  double rr = r*r;
215  double small = -0.5 + 0.3333333333333333333333*r;
216  double large = -0.25 + 0.2*r;
217  double rrrr = rr * rr;
218  small *= rr;
219  small += r;
220  large *= rrrr;
221  r = small + large;
222  log2x += r * recip_ln2;
223  }
224 
225  // multiply by Y
226  double ylog2x = y * log2x;
227 
228 // now we need to calculate 2**ylog2x
229 
230  //deal with overflow
231  if( ylog2x >= 128.0 )
232  return (float) (0x1.0p128 * ylog2x); //set overflow and return inf
233 
234  //deal with underflow
235  if( ylog2x <= -150.0 )
236  return (float) ( ylog2x * 0x1.0p-1022 ); //minimum y * maximum log2(x) is ~-1.0p128 * ~128 = -1.0p135, so we can be sure that we'll drive this to underflow
237 
238  //separate ylog2x into integer and fractional parts
239  int exp = (int) ylog2x;
240  double f = ylog2x - exp; //may be negative
241 
242  // Calculate 2**fract
243  // 8th order minimax fit of exp2 on [-1.0,1.0]. |error| < 0.402865722354948566583852e-9:
244  static const double c0 = 1.0 + 0.278626872016317130037181614004e-10;
245  static const double c1 = .693147176943623740308984004029708;
246  static const double c2 = .240226505817268621584559118975830;
247  static const double c3 = 0.555041568519883074165425891257052e-1;
248  static const double c4overc8 = 0.961813690023115610862381719985771e-2 / 0.134107709538786543922336536865157e-5;
249  static const double c5overc8 = 0.133318252930790403741964203236548e-2 / 0.134107709538786543922336536865157e-5;
250  static const double c6overc8 = 0.154016177542147239746127455226575e-3 / 0.134107709538786543922336536865157e-5;
251  static const double c7overc8 = 0.154832722143258821052933667742417e-4 / 0.134107709538786543922336536865157e-5;
252  static const double c8 = 0.134107709538786543922336536865157e-5;
253 
254  double z = 1.0;
255  if( 0.0 != f )
256  { // don't set inexact if we don't need to
257  double ff = f * f;
258  double s7 = c7overc8 * f; double s3 = c3 * f;
259  double s5 = c5overc8 * f; double s1 = c1 * f;
260  double ffff = ff * ff;
261  s7 += c6overc8; s3 += c2;
262  s5 += c4overc8; s1 += c0;
263  s7 *= ff; s3 *= ff;
264  s5 += ffff;
265  double c8ffff = ffff * c8;
266  s7 += s5; s3 += s1;
267  s7 *= c8ffff;
268  z = s3 + s7;
269  }
270 
271 
272  //prepare 2**i
273  union{ uint64_t u; double d; } two_exp = { ((uint64_t) exp + 1023) << 52 };
274 
275  return (float) (z * two_exp.d );

279  // and one special case -- call sqrt for |y| == 0.5

289  y = y < 0x1.fffffep30f ? y : 0x1.fffffep30f;
