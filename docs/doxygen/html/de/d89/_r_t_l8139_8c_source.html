91 
92  RTL8139_ret = pci_read( // Ищем среди устройств
93  pci_get_device(RTL8139_VENDOR_ID, RTL8139_ret_ID, -1),
94  PCI_BAR0
95  );
96 
97  if(RTL8139_ret == 0) {
98  qemu_log("RTL8139 can't install");
99  return -1;
100  }
101 
102  pci_RTL8139_device = pci_get_device(RTL8139_VENDOR_ID, RTL8139_ret_ID, -1);
103 
104  RTL8139_device.bar_type = RTL8139_ret & 0x1;
105 
106  RTL8139_device.io_base = RTL8139_ret & (~0x3); // Получаем базу ввода-вывода или базу памяти, извлекая старшие 28/30 бит
107  RTL8139_device.mem_base = RTL8139_ret & (~0xf);
108  tty_printf("RTL8139 use %s, base: %x\n", (RTL8139_device.bar_type == 0)? "mem based":"port based", (RTL8139_device.bar_type != 0)?RTL8139_device.io_base:RTL8139_device.mem_base);
109

111 
112  // Включаем для PCI
113  uint32_t pci_command_reg = pci_read(pci_RTL8139_device, PCI_COMMAND);
114  if(!(pci_command_reg & (1 << 2))) {
115  pci_command_reg |= (1 << 2);
116  pci_write(pci_RTL8139_device, PCI_COMMAND, pci_command_reg);
117  }
118  // Отправляем 0x00 в регистр CONFIG_1 (0x52), чтобы установить LWAKE + LWPTN как активный. (Включаем)
119  outb(RTL8139_device.io_base + 0x52, 0x0);
120

122

129  outs(RTL8139_device.io_base + 0x3C, 0x0005);

132  outl(RTL8139_device.io_base + 0x44, 0xf | (1 << 7));
