31void RTL8139_read_mac_addr() {

33 RTL8139_raw_mac[0] = inl(RTL8139_device.io_base + 0x00);
34 RTL8139_raw_mac[1] = ins(RTL8139_device.io_base + 0x04);
35
36 RTL8139_device.mac_addr[0] = RTL8139_raw_mac[0] >> 0;
37 RTL8139_device.mac_addr[1] = RTL8139_raw_mac[0] >> 8;

39 RTL8139_device.mac_addr[3] = RTL8139_raw_mac[0] >> 24;
40
41 RTL8139_device.mac_addr[4] = RTL8139_raw_mac[1] >> 0;
42 RTL8139_device.mac_addr[5] = RTL8139_raw_mac[1] >> 8;
43
44 tty_printf("MAC Address: %v:%v:%v:%v:%v:%v\n",

46 RTL8139_device.mac_addr[3], RTL8139_device.mac_addr[4], RTL8139_device.mac_addr[5]

88// Настраивает RTL8139
89int RTL8139_init() {
90 outb(RTL8139_io_addres + 0x52, 0x0); // Включаем RTL
91
92 RTL8139_ret = pci_read( // Ищем среди устройств
93 pci_get_device(RTL8139_VENDOR_ID, RTL8139_ret_ID, -1),
94 PCI_BAR0
95 );
96
97 if(RTL8139_ret == 0) {
98 qemu_log("RTL8139 can't install");
99 return -1;
100 }
101
102 pci_RTL8139_device = pci_get_device(RTL8139_VENDOR_ID, RTL8139_ret_ID, -1);
103
104 RTL8139_device.bar_type = RTL8139_ret & 0x1;
105
106 RTL8139_device.io_base = RTL8139_ret & (~0x3); // Получаем базу ввода-вывода или базу памяти, извлекая старшие 28/30 бит
107 RTL8139_device.mem_base = RTL8139_ret & (~0xf);
108 tty_printf("RTL8139 use %s, base: %x\n", (RTL8139_device.bar_type == 0)? "mem based":"port based", (RTL8139_device.bar_type != 0)?RTL8139_device.io_base:RTL8139_device.mem_base);
109
110 RTL8139_device.tx_cur = 0; // Устанавливаем текущий TSAD
111
112 // Включаем для PCI
113 uint32_t pci_command_reg = pci_read(pci_RTL8139_device, PCI_COMMAND);
114 if(!(pci_command_reg & (1 << 2))) {
115 pci_command_reg |= (1 << 2);
116 pci_write(pci_RTL8139_device, PCI_COMMAND, pci_command_reg);
117 }
118 // Отправляем 0x00 в регистр CONFIG_1 (0x52), чтобы установить LWAKE + LWPTN как активный. (Включаем)
119 outb(RTL8139_device.io_base + 0x52, 0x0);
120
121 RTL8139_reset(); // Перезапуск
122
123 // Выделяем буффер
124 RTL8139_device.rx_buffer = kheap_malloc(8192 + 16 + 1500);
125 memset(RTL8139_device.rx_buffer, 0x0, 8192 + 16 + 1500);
126 outl(RTL8139_device.io_base + 0x30, (uint32_t)kv2p(RTL8139_device.rx_buffer));
127
128 // Устанавливает биты TOK и ROK в высокий уровень
129 outs(RTL8139_device.io_base + 0x3C, 0x0005);
130
131 // (1 << 7) — это бит WRAP, 0xf — это AB+AM+APM+AAP. Ps. я ничего не понял тут
132 outl(RTL8139_device.io_base + 0x44, 0xf | (1 << 7));
133
134 // Устанавливает биты RE и TE в высокий уровень
135 outb(RTL8139_device.io_base + 0x37, 0x0C);
136 RTL8139_read_mac_addr();
137
138 // Регистрация прерываний
139 uint32_t irq_num = pci_read(pci_RTL8139_device, PCI_INTERRUPT_LINE);
140 register_interrupt_handler(32 + irq_num, &RTL8139_handler);
141 IRQ_clear_mask(irq_num);
142 qemu_log("RTL8139 installed, idt: %d, irq: %d", 32 + irq_num, irq_num);
143
144
145 return 1;
146}
