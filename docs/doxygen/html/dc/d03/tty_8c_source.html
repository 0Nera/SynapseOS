161  uint8_t *write_ptr = (uint8_t*) back_framebuffer_addr;
162  uint32_t num_bytes = (framebuffer_pitch * VESA_HEIGHT) - (framebuffer_pitch * (num_rows * 17));
163  memcpy(write_ptr, read_ptr, num_bytes);
164 
165  // Очистка строк

167  memset(write_ptr, 0, framebuffer_pitch * (num_rows * 17));
168
